add_library(mltvmfx MODULE
  factory.c
  filter_chroma_hold.c
  filter_chroma.c
  filter_mono.c
  filter_shape.c

)

  filter_chroma_hold.yml

  filter_mono.yml
  filter_shape.yml
  producer_pgm.yml
  DESTINATION ${MLT_INSTALL_DATA_DIR}/vmfx
